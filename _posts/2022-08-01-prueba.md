function IpVictim(){

	ip_victim=$1

	timeout 1 bash -c "ping -c 1 $ip_victim > /dev/null 2>&1"

	if [ $(echo $?) == "0" ]; then

		echo "$ip_victim" > /home/dansh/.config/bin/ip.txt
	else

		echo "Disabled" > /home/dansh/.config/bin/ip.txt
	fi

}


parameter_counter=0


while getopts "e:i:h:" arg; do

	case $arg in

		e) explorationMode=$OPTARG; let parameter_counter+=1;;
		i) ip_victim=$OPTARG; let parameter_counter+=1;;
		h) helpPanel;;

	esac

done


if [ $parameter_counter == "0" ]; then

	helpPanel

elif [ $explorationMode == "ip" ]; then

	IpVictim $ip_victim

fi
